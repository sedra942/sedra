        let isAuthenticated = false;
        let isUserAuthenticated = false;

        let currentVideo = null;

        // Admin credentials (in production, this would be server-side authenticated)
        const ADMIN_USERNAME = 'admin';
        const ADMIN_PASSWORD = 'sedra2023';
        
        // Switch view function
        function switchView(view, params) {
            currentView = view;
            document.getElementById('main-view').style.display = view === 'main' ? 'block' : 'none';
            document.getElementById('login-view').style.display = view === 'login' ? 'block' : 'none';
            document.getElementById('admin-view').style.display = view === 'admin' ? 'block' : 'none';
            document.getElementById('video-view').style.display = view === 'video' ? 'block' : 'none';
            document.getElementById('user-login-view').style.display = view === 'user-login' ? 'block' : 'none';
            document.getElementById('payment-view').style.display = view === 'payment' ? 'block' : 'none';
            document.getElementById('user-dashboard-view').style.display = view === 'user-dashboard' ? 'block' : 'none';
            
            if (view === 'video' && params) {
                currentVideo = params;
                loadVideo(params);
            }
            
            if (view === 'user-login' && params) {
                selectedPlan = params;
                document.getElementById('selected-plan-name').textContent = params.name;
                document.getElementById('selected-plan-price').textContent = params.price;
            }
        }
        
        // Load video function
        function loadVideo(videoData) {
            document.getElementById('video-title').innerText = videoData.title;
            document.getElementById('video-subject').innerText = videoData.subject;
            
            // Check if this is premium content and user is not authenticated
            if (videoData.subject !== 'الرياضيات' && !isUserAuthenticated) {
                document.getElementById('premium-content-overlay').classList.remove('hidden');
                document.getElementById('video-content').classList.add('hidden');
            } else {
                document.getElementById('premium-content-overlay').classList.add('hidden');
                document.getElementById('video-content').classList.remove('hidden');
            }
        }
        
        // Admin login function
        function attemptLogin() {
            const username = document.getElementById('username').value;
            const password = document.getElementById('password').value;
            
            if (username === ADMIN_USERNAME && password === ADMIN_PASSWORD) {
                isAuthenticated = true;
                switchView('admin');
                document.getElementById('login-error').style.display = 'none';
            } else {
                document.getElementById('login-error').style.display = 'block';
            }
        }
        
        // User login with Google
        function simulateGoogleLogin() {
            // Show loading state
            document.getElementById('google-login-button').classList.add('hidden');
            document.getElementById('google-login-loading').classList.remove('hidden');
            
            // Simulate network delay
            setTimeout(() => {
                isUserAuthenticated = true;
                userEmail = document.getElementById('user-email').value;
                if (!userEmail) {
                    userEmail = 'user@gmail.com';
                }
                
                // Update UI to show logged in state
                document.getElementById('google-login-loading').classList.add('hidden');
                document.getElementById('google-login-success').classList.remove('hidden');
                
                // Navigate to payment page after a short delay
                setTimeout(() => {
                    switchView('payment');
                }, 1000);
            }, 1500);
        }
        
        // Process payment
        function processPayment() {
            // Show loading state
            document.getElementById('payment-button').classList.add('hidden');
            document.getElementById('payment-loading').classList.remove('hidden');
            
            // Simulate network delay for payment processing
            setTimeout(() => {
                // Show success message
                document.getElementById('payment-loading').classList.add('hidden');
                document.getElementById('payment-success').classList.remove('hidden');
                
                // Navigate to user dashboard after payment
                setTimeout(() => {
                    switchView('user-dashboard');
                }, 1500);
            }, 2000);
        }
        
        // Select a plan for purchase

            if (!isUserAuthenticated) {
                switchView('user-login', { name, price });
            } else {
                selectedPlan = { name, price };
                switchView('payment');
            }
        }

        // Submit comment function
        function submitComment(event) {
            event.preventDefault();
            const commentText = document.getElementById('comment-text').value;
            
            if (commentText.trim() === '') return;
            
            // Add comment to the list
            const commentsList = document.getElementById('comments-list');
            const newComment = document.createElement('div');
            newComment.className = 'bg-gray-100 dark:bg-gray-700 rounded-lg p-4 mb-4';
            
            newComment.innerHTML = `
                <div class="flex items-center mb-2">
                    <div class="w-10 h-10 bg-primary text-white rounded-full flex items-center justify-center font-bold text-xl">ط</div>
                    <div class="mr-3">
                        <h4 class="font-semibold">طالب</h4>
                        <p class="text-gray-500 dark:text-gray-400 text-xs">الآن</p>
                    </div>
                </div>
                <p class="text-gray-600 dark:text-gray-300">${commentText}</p>
            `;
            
            commentsList.prepend(newComment);
            
            // Clear input
            document.getElementById('comment-text').value = '';
        }
        
        // Toggle chatbot
        function toggleChatbot() {
            chatbotOpen = !chatbotOpen;
            const chatbot = document.getElementById('chatbot');
            const chatbotButton = document.getElementById('chatbot-button');
            
            if (chatbotOpen) {
                chatbot.classList.remove('translate-y-full');
                chatbot.classList.add('translate-y-0');
                chatbotButton.classList.add('hidden');
            } else {
                chatbot.classList.remove('translate-y-0');
                chatbot.classList.add('translate-y-full');
                chatbotButton.classList.remove('hidden');
            }
        }
        
        // Send chatbot message
        function sendChatbotMessage(event) {
            event.preventDefault();
            const messageInput = document.getElementById('chatbot-message');
            const message = messageInput.value;
            
            if (message.trim() === '') return;
            
            // Add user message
            const chatMessages = document.getElementById('chatbot-messages');
            const userMessage = document.createElement('div');
            userMessage.className = 'flex justify-end mb-3';
            userMessage.innerHTML = `
                <div class="bg-primary text-white rounded-lg py-2 px-4 max-w-[80%]">
                    <p>${message}</p>
                </div>
            `;
            chatMessages.appendChild(userMessage);
            
            // Clear input
            messageInput.value = '';
            
            // Scroll to bottom
            chatMessages.scrollTop = chatMessages.scrollHeight;
            
            // Simulate bot response after a short delay
            setTimeout(() => {
                const botMessage = document.createElement('div');
                botMessage.className = 'flex mb-3';
                botMessage.innerHTML = `
                    <div class="bg-gray-200 dark:bg-gray-700 rounded-lg py-2 px-4 max-w-[80%]">
                        <p>شكراً لتواصلك معنا! سيتم الرد عليك في أقرب وقت ممكن. يمكنك أيضاً التواصل معنا مباشرة عبر واتساب على الرقم المذكور في صفحة الاتصال.</p>
                    </div>
                `;
                chatMessages.appendChild(botMessage);
                
                // Scroll to bottom
                chatMessages.scrollTop = chatMessages.scrollHeight;
            }, 1000);
        }
